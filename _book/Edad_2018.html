14.3.1 Tibble Agrupado por Entidad Federativa y Sexo:

  mutate(Casos = as.numeric(Casos))

edades_2018_GPENT <- edades_2018%>%
  group_by(entidad, Partido, Rango_Edad)%>%
  reframe(Casos = sum(Casos))
